mesh2get <- function(x, y){
    p <- (y * 60) %/% 40
    a <- (y * 60) %% 40
    u <- (x - 100) %/% 1
    f <- (x - 100) %% 1
    q <- a %/% 5
    b <- a %% 5
    v <- (f * 60) %/% 7.5
    g <- (f * 60) %% 7.5

    p * 10000 + u * 100 + q * 10 + v * 1     
}

latget <- function(mesh2){
    trunc(mesh2 / 10000) * 2/3 + trunc(mesh2 %% 100 / 10) * 2/3 / 8
}
longet <- function(mesh2){
    100 + trunc((mesh2 - trunc(mesh2 / 10000)*10000) / 100) + mesh2 %% 10 / 8
}
