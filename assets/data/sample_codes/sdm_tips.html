mesh2get <- function(x, y){
    p <- (y * 60) %/% 40
    a <- (y * 60) %% 40
    u <- (x - 100) %/% 1
    f <- (x - 100) %% 1
    q <- a %/% 5
    b <- a %% 5
    v <- (f * 60) %/% 7.5
    g <- (f * 60) %% 7.5

    p * 10000 + u * 100 + q * 10 + v * 1     
}

latget <- function(mesh2){
    trunc(mesh2 / 10000) * 2/3 + trunc(mesh2 %% 100 / 10) * 2/3 / 8
}
longet <- function(mesh2){
    100 + trunc((mesh2 - trunc(mesh2 / 10000)*10000) / 100) + mesh2 %% 10 / 8
}

モデルの評価¶

In [21]:
plotROC(model)

↑
x軸：在予測されたメッシュのうち、「そのメッシュには在記録はない」割合
「在記録がない」を疑似的に不在メッシュとみなして値を算出しているが、必ずしもそのようなメッシュで「不在」であると解釈されるわけではない。

y軸：在予測されたメッシュのうち、「そのメッシュに在記録がある」割合

閾値を1つ決めると、点が1つ得られる。曲線は、閾値を様々に変化させたときの軌跡

右上の(0,1)点が、最もパフォーマンスが高い場合を示す。スキル統計量最大化による閾値選択では、ROC上で(0,1)に最も近い点に対応する閾値を選択する。

AUCは、ROCの下側の面積の大きさを表す。1.0に近いほど、モデルの性能が良いことを示す。
